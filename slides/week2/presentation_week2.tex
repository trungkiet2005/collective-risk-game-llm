\section{Tuần này làm gì}

\begin{frame}{Nhắc lại: tuần trước ba model nhỏ chơi khác hẳn con người}
  \small
  \begin{block}{Ba điểm bất thường ở điều kiện baseline}
    \begin{itemize}\setlength\itemsep{2pt}
      \item \textbf{Không phản ứng với rủi ro}: dù thảm hoạ ở mức 90\%, 50\% hay 10\%, cả ba model vẫn đạt gần 100\% (tỉ lệ của người giảm dần 50 $\to$ 10 $\to$ 0\%).
      \item \textbf{Càng cuối game càng hợp tác}, không ``phản bội'' ở vòng chót --- ngược hẳn với người.
      \item \textbf{Luôn đóng vượt mục tiêu}: Qwen-7B $\sim$233, Llama-8B $\sim$185, Gemma-9B $\sim$129 (trên 240).
    \end{itemize}
  \end{block}
  \begin{exampleblock}{Hai câu hỏi còn bỏ ngỏ $\Rightarrow$ hai thí nghiệm tuần này}
    \begin{enumerate}\setlength\itemsep{2pt}
      \item Các bất thường này có phải chỉ vì model còn \emph{nhỏ}? $\to$ thử \alert{model lớn hơn}.
      \item Model \emph{không hiểu luật}, hay hiểu mà vẫn phớt lờ rủi ro? $\to$ \alert{kiểm tra đọc-hiểu}.
    \end{enumerate}
  \end{exampleblock}
\end{frame}

\section{Mở rộng 1 --- Model lớn (scaling)}

\begin{frame}{Cách làm và điều ta muốn biết}
  \begin{block}{Thao tác}
    {\footnotesize Giữ nguyên mọi thứ của tuần trước --- prompt baseline trung tính, ba mức rủi ro, hai ngôn ngữ, mười lần lặp --- chỉ \textbf{đổi sang model lớn hơn}: Qwen2.5 từ 7B lên 32B, Gemma-2 từ 9B lên 27B. Llama-3.1-8B giữ nguyên làm mốc so sánh (chưa có bản Llama lớn nên để trống cặp này).}
  \end{block}
  \begin{exampleblock}{Hai khả năng đối nghịch}
    \begin{itemize}
      \item \emph{Nếu do model nhỏ}: model lớn sẽ biết ``sợ'' rủi ro hơn và chơi giống người hơn.
      \item \emph{Nếu là đặc tính chung}: model lớn vẫn phớt lờ rủi ro, chỉ thay đổi ở khâu \emph{tiêu tiền}.
    \end{itemize}
  \end{exampleblock}
\end{frame}

\section{Mở rộng 2 --- Kiểm tra đọc-hiểu}

\begin{frame}{Đo đọc-hiểu bằng cách nào?}
  \begin{block}{Cách hỏi --- xen kẽ ngay trong lúc chơi (in-situ)}
    {\footnotesize Trong khi model chơi ván baseline bình thường, mỗi vòng tôi gửi \textbf{thêm} một loạt câu hỏi về chính ván đó. Các câu này gửi qua một lượt \emph{riêng}, không nằm trong prompt quyết định, nên \textbf{không làm đổi nước đi} --- đã kiểm chứng: bật hay tắt probe thì quyết định vẫn giống hệt nhau.}
  \end{block}
  \begin{block}{Cách chấm --- so với đáp án đúng của engine}
    {\footnotesize Mỗi câu có đáp án đúng tính sẵn từ trạng thái game (ground truth), nên chấm đúng/sai hoàn toàn tự động. Khoảng \textbf{54.500 câu mỗi model}.}
  \end{block}
  \begin{exampleblock}{Ba nhóm câu hỏi}
    {\footnotesize \textbf{Rules} --- luật tĩnh (đọc thẳng từ đề) \quad$\bullet$\quad \textbf{Time} --- tra lịch sử từng vòng \quad$\bullet$\quad \textbf{State} --- thống kê tích luỹ (phần lớn phải \emph{tự cộng dồn}).}
  \end{exampleblock}
\end{frame}

\begin{frame}{Bộ câu hỏi thật --- 20 câu mỗi lượt}
  {\footnotesize Nhãn: \rd{} = đáp án \emph{in sẵn} trong prompt, chỉ cần đọc \quad $\vert$ \quad \sm{} = phải \emph{tự tính} từ lịch sử.}
  \smallskip
  \begin{columns}[T]
    \begin{column}{0.49\textwidth}
      {\scriptsize\color{deepteal}\textbf{RULES} (8 câu --- toàn bộ \rd)}\par
      {\scriptsize
      \begin{itemize}\setlength\itemsep{1pt}
        \item Mỗi vòng được đóng những mức nào? \;(0/2/4)
        \item Vốn ban đầu mỗi người? \;(40)
        \item Mục tiêu chung cả game? \;(120)
        \item Game dài bao nhiêu vòng? \;(10)
        \item Trượt mục tiêu thì xác suất mất hết là mấy \%? \;(90/50/10)
        \item Trượt mà gặp thảm hoạ, bạn còn bao nhiêu? \;(0)
        \item Mức đóng tối đa / tối thiểu một vòng? \;(4 / 0)
      \end{itemize}}
      \smallskip
      {\scriptsize\color{deepteal}\textbf{TIME} (4 câu --- tra lịch sử)}\par
      {\scriptsize
      \begin{itemize}\setlength\itemsep{1pt}
        \item Đang ở vòng thứ mấy? \rd
        \item Vòng $i$, người ghế P$x$ / chính bạn đóng bao nhiêu? \rd
        \item Vòng $i$, cả 6 người đóng tổng bao nhiêu? \sm
      \end{itemize}}
    \end{column}
    \begin{column}{0.49\textwidth}
      {\scriptsize\color{deepteal}\textbf{STATE} (8 câu --- thống kê tích luỹ)}\par
      {\scriptsize
      \begin{itemize}\setlength\itemsep{1pt}
        \item Tổng quỹ chung đến giờ là bao nhiêu? \sm
        \item Còn thiếu bao nhiêu nữa để đạt 120? \sm
        \item Ghế P$x$ đã đóng tổng cộng bao nhiêu? \sm
        \item Chính bạn đã đóng tổng cộng bao nhiêu? \sm
        \item Bạn còn lại bao nhiêu vốn chưa đóng? \rd
        \item Ghế P$x$ đã đóng đúng mức 4 mấy lần? \sm
        \item Còn mấy vòng nữa (tính cả vòng này)? \sm
        \item Nhóm đã đạt mục tiêu chưa? (có/không) \sm
      \end{itemize}}
      \smallskip
      \begin{block}{Ý đồ thiết kế}
        {\scriptsize Rules/Time chủ yếu là \emph{đọc}; State chủ yếu là \emph{cộng dồn}. Tách hai loại để biết model yếu ở khâu \emph{hiểu luật} hay khâu \emph{tính toán trạng thái}.}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Một probe hoàn chỉnh --- đúng những gì model nhận được}

\begin{frame}[fragile]{Ví dụ trả lời thật --- biết luật, nhưng không cộng được}

\section{Tổng kết tuần 2}

\begin{frame}{Hai mảnh ghép vào bức tranh chung}
  \begin{block}{Hợp tác của LLM là \emph{thói quen có sẵn}, không phải \emph{tính toán}}
    \begin{itemize}
      \item Phần đọc-hiểu cho thấy: model \textbf{hiểu luật} và \textbf{biết rủi ro} (Rules $\sim$100\%, câu rủi ro $\sim$100\%), nhưng \textbf{không cộng được trạng thái} (chỉ 10--21\% khi phải tự tính).
      \item Hệ quả: model vẫn đạt mục tiêu \emph{dù} theo dõi sai số quỹ $\Rightarrow$ hợp tác là thiên hướng có sẵn, không phải suy luận theo trạng thái.
      \item Phần scaling cho thấy: phóng to chỉ cắt phần \emph{đóng dư} (Qwen 233 $\to$ 128) --- model \emph{tiêu khôn hơn}, nhưng vẫn phớt lờ rủi ro và vẫn tăng dần.
    \end{itemize}
  \end{block}
  \take{Phớt lờ rủi ro không phải do \emph{hiểu sai} (đọc-hiểu đã bác bỏ), cũng không phải do \emph{model nhỏ} (scaling đã bác bỏ) $\Rightarrow$ nó là một \alert{thiên hướng hành vi} của lớp model này.}
\end{frame}

\begin{frame}{So với tuần trước: cập nhật bảng tổng}
  \centering\small
  \begin{tabular}{lll}
    \toprule
    Câu hỏi & Tuần trước & Bằng chứng tuần này \\
    \midrule
    Có phải do model nhỏ?      & (chỉ là giả thuyết) & \alert{Không} --- 27/32B vẫn phẳng 100\% \\
    Model có hiểu luật không?  & (chưa đo)           & \alert{Có} --- Rules đúng 94--100\% \\
    Model có biết mức rủi ro?  & (chưa đo)           & \alert{Có} --- câu rủi ro đúng $\sim$100\% \\
    Vì sao theo dõi quỹ sai?   & ``hợp tác là thói quen'' & \alert{Không cộng dồn được} (State 10--21\%) \\
    Trục ngôn ngữ nằm ở đâu?   & ở đóng góp (Llama)  & + \alert{số học trạng thái} (State $-19$đ\% ở VN) \\
    \bottomrule
  \end{tabular}
