\begin{frame}{Đo đọc-hiểu bằng cách nào?}
  \begin{block}{Hỏi --- xen kẽ trong lúc chơi (in-situ)}
    Mỗi vòng hỏi thêm về chính ván đó, qua lượt \emph{riêng} $\Rightarrow$ \textbf{không đổi nước đi} (đã kiểm chứng).
  \end{block}
  \begin{block}{Chấm --- so đáp án engine}
    Ground-truth tính sẵn, đúng/sai tự động. \textbf{$\sim$54.500 câu/model}; nay đo tới \textbf{70--72B} (Qwen-32B/72B, Gemma-27B, Llama-70B\footnote{\scriptsize bản đo comprehension là Llama-3.3-70B; baseline hành vi là Llama-3.1-70B}) để xem scaling có ``chữa'' được không.
  \end{block}
  \begin{exampleblock}{Ba nhóm câu}
    \textbf{Rules} đọc luật \quad$\bullet$\quad \textbf{Time} tra lịch sử \quad$\bullet$\quad \textbf{State} cộng dồn
  \end{exampleblock}
\end{frame}

\section{Tổng kết Turn 2}

\begin{frame}{Ba mảnh ghép vào bức tranh chung}
  \begin{block}{Hợp tác của LLM là \emph{thói quen}, không phải \emph{tính toán}}
    \begin{itemize}\setlength\itemsep{3pt}
      \item Hiểu luật + biết rủi ro (\good{$\sim$100\%}); cộng-dồn trạng thái yếu ở model nhỏ nhưng \emph{hồi gần hết} khi phóng to (pool ẩn 14$\to$\good{88\%} ở Qwen-72B)
      \item $\Rightarrow$ tới 70--72B model \emph{cộng được} trạng thái \& biết rủi ro mà \good{vẫn} phớt lờ --- hợp tác là thiên hướng, không phải suy luận theo trạng thái
      \item Scaling chỉ cắt phần đóng dư (233$\to$128): tiêu khôn hơn, vẫn phớt lờ rủi ro tới 72B
      \item Model frontier \textbf{đóng} (Gemini) cũng \good{phớt lờ rủi ro} + đóng vừa đủ $\Rightarrow$ bất thường không riêng lớp model mở/tự host
      \item Llama-70B: model lớn đầu tiên \weak{thất bại thật} --- nhưng do \alert{ngôn ngữ} (VN hụt vạch), không do rủi ro
    \end{itemize}
  \end{block}
  \take{Phớt lờ rủi ro: không do \emph{hiểu sai}, không do \emph{model nhỏ}, không do \emph{mở/đóng} $\Rightarrow$ \alert{thiên hướng hành vi} của lớp model này.}
\end{frame}

\begin{frame}{Điểm thảo luận với thầy}
  \textbf{Scaling} (đã có đủ ba họ + 70/72B)
  \begin{itemize}\setlength\itemsep{2pt}
    \item Llama-70B \good{đã chạy}: lộ điểm gãy \alert{ngôn ngữ} (VN 112$<$120, thảm hoạ 73\%) --- đào sâu vì sao chỉ Llama?
    \item Qwen-32B tụt nhẹ tỉ lệ đạt (97\%) vì chơi sát vạch --- lấy \emph{đóng góp} làm biến chính?
  \end{itemize}
  \smallskip
  \textbf{Đọc-hiểu} (đã đo tới 70--72B)
  \begin{itemize}\setlength\itemsep{2pt}
    \item Scaling hồi cộng-dồn \good{gần hết} (pool ẩn 14$\to$88\% ở Qwen-72B) --- nhưng hành vi phớt lờ rủi ro \emph{không} đổi $\Rightarrow$ tách bạch được ``hiểu'' vs ``hành vi''.
    \item Đưa \emph{tổng quỹ in sẵn} vào prompt quyết định --- hợp tác có đổi khi được ``nhắc''?
  \end{itemize}
\end{frame}
